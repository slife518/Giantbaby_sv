use a01


CREATE TABLE IF NOT EXISTS `ci_sessions` (
        `id` varchar(128) NOT NULL,
        `ip_address` varchar(45) NOT NULL,
        `timestamp` int(10) unsigned DEFAULT 0 NOT NULL,
        `data` blob NOT NULL,
        KEY `ci_sessions_timestamp` (`timestamp`)
);
ALTER TABLE ci_sessions ADD PRIMARY KEY (id);


CREATE TABLE record (
    id  int(11) NOT NULL AUTO_INCREMENT,
    baby_id int(11) NOT NULL,
    record_date date NOT NULL,
    record_time time NOT NULL,
    milk varchar(3),
    rice varchar(3),
    author varchar(50),
    description  text NULL ,
    updated  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created  datetime DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
);

CREATE TABLE user (
  email varchar(50) NOT NULL,
  password varchar(255) NOT NULL,
  nickname varchar(20),
  tel varchar(15),
  address1 varchar(50),
  address2 varchar(50),
  created datetime NOT NULL,
  updated datetime,
  PRIMARY KEY (email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;



CREATE TABLE relation (
baby_id int(11) NOT NULL,
email varchar(50) NOT NULL,
approval varchar(5) DEFAULT '0',
level int(3),
created  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
updated datetime,
PRIMARY KEY (baby_id, email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


CREATE TABLE baby (
baby_id int(11) NOT NULL AUTO_INCREMENT,
birthday varchar(6),
babyname varchar(13),
sex varchar(1),
father varchar(20),
mother varchar(20),
owner varchar(20),
created  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
updated datetime,
PRIMARY KEY (baby_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE environment (
  email varchar(50) NOT NULL,
  font varchar(50),
  size varchar(4),
  created  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated datetime,
  PRIMARY KEY (email)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8;

  CREATE TABLE board (
      id  int(11) NOT NULL AUTO_INCREMENT,
      writer varchar(20),
      title varchar(30),
      content text NULL,
      registDay  datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
      PRIMARY KEY (id)
  );

  ALTER TABLE `baby` ADD `father` VARCHAR(20) NULL AFTER `sex`, ADD `mother` VARCHAR(20) NULL AFTER `father`, ADD `owner` VARCHAR(50) NOT NULL AFTER `mother`;
  ALTER TABLE `relation` ADD `level` INT(3) NULL AFTER `email`;
